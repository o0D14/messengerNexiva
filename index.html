    <script>

        // 1. ПОДКЛЮЧАЕМСЯ К ТВОЕМУ СЕРВЕРУ

        const socket = io('http://192.168.1.102:3000');

        

        let myName = "";

        let myUsername = "";

        let currentChatUser = null; // С кем сейчас открыт чат

        let currentRoomId = "";     // ID комнаты

        let allMessages = [];       // Все сообщения с сервера


        // Логика входа

        function login() {

            const nameInput = document.getElementById('input-name').value.trim();

            const usernameInput = document.getElementById('input-username').value.trim();

            

            if (!nameInput || !usernameInput) {

                alert("Заполните оба поля!");

                return;

            }

            

            myName = nameInput;

            myUsername = usernameInput.startsWith('@') ? usernameInput : '@' + usernameInput;

            

            document.getElementById('login-screen').style.display = 'none';

            document.getElementById('app').style.display = 'flex';

            

            // Сообщаем серверу, что веб-клиент вошел в сеть!

            socket.emit('user_joined', { name: myName, username: myUsername });

        }


        // 2. СЛУШАЕМ СЕРВЕР


        // Сервер прислал список людей в сети

        socket.on('online_users', (users) => {

            const list = document.getElementById('users-list');

            list.innerHTML = ''; // Очищаем список

            

            users.forEach(user => {

                // Себя не показываем

                if (user.username === myUsername) return;

                

                const li = document.createElement('li');

                li.className = 'user-item';

                // Если этот юзер сейчас открыт, подсвечиваем его

                if (currentChatUser && currentChatUser.username === user.username) {

                    li.classList.add('active');

                }

                

                li.innerHTML = `

                    <div class="online-dot"></div>

                    <div>

                        <div style="font-weight: bold;">${user.name}</div>

                        <div style="font-size: 12px; color: #888;">${user.username}</div>

                    </div>

                `;

                

                // При клике на юзера - открываем с ним чат

                li.onclick = () => openChat(user);

                list.appendChild(li);

            });

        });


        // Сервер прислал историю

        socket.on('load_history', (history) => {

            allMessages = history;

            if (currentChatUser) renderMessages();

        });


        // Сервер прислал новое сообщение (от кого угодно)

        socket.on('receive_message', (msg) => {

            // Игнорируем системные команды (звонки, заморозки)

            if (msg.text.startsWith('SYS_')) return;

            

            // Добавляем в общую базу

            if (!allMessages.find(m => m.id === msg.id)) {

                allMessages.push(msg);

                

                // Если мы сейчас сидим в этом чате - перерисовываем

                if (currentChatUser && msg.chatId === currentRoomId) {

                    renderMessages();

                    

                    // Звук уведомления на компе (если страница открыта)

                    if (msg.senderId !== myUsername) {

                        new Audio('https://actions.google.com/sounds/v1/alarms/beep_short.ogg').play().catch(()=>{});

                    }

                }

            }

        });


        // 3. ЛОГИКА ИНТЕРФЕЙСА


        // Функция генерации ID чата (ТАКАЯ ЖЕ КАК В SWIFT!)

        function generatePrivateChatId(user1, user2) {

            return [user1, user2].sort().join('_');

        }


        function openChat(user) {

            currentChatUser = user;

            currentRoomId = generatePrivateChatId(myUsername, user.username);

            

            document.getElementById('empty-state').style.display = 'none';

            document.getElementById('chat-area').style.display = 'flex';

            document.getElementById('chat-title').innerText = user.name;

            

            // Подсвечиваем в списке

            document.querySelectorAll('.user-item').forEach(el => el.classList.remove('active'));

            event.currentTarget.classList.add('active');

            

            renderMessages();

        }


        function renderMessages() {

            const container = document.getElementById('messages-container');

            container.innerHTML = '';

            

            // Фильтруем сообщения только для текущей комнаты

            const chatMsgs = allMessages.filter(m => m.chatId === currentRoomId && !m.text.startsWith('SYS_'));

            

            chatMsgs.forEach(msg => {

                const isMine = msg.senderId === myUsername;

                const div = document.createElement('div');

                div.className = `msg ${isMine ? 'msg-mine' : 'msg-other'}`;

                

                const date = new Date(msg.timestamp);

                const time = `${date.getHours().toString().padStart(2, '0')}:${date.getMinutes().toString().padStart(2, '0')}`;

                

                div.innerHTML = `

                    <div>${msg.text} ${msg.effect ? msg.effect : ''}</div>

                    <div class="msg-time">${time}</div>

                `;

                container.appendChild(div);

            });

            

            // Прокрутка вниз

            container.scrollTop = container.scrollHeight;

        }


        // 4. ОТПРАВКА СООБЩЕНИЯ

        function sendMessage() {

            const input = document.getElementById('msg-input');

            const text = input.value.trim();

            

            if (!text || !currentChatUser) return;

            

            const newId = crypto.randomUUID();

            const timestamp = new Date().toISOString();

            

            const data = {

                id: newId,

                chatId: currentRoomId,

                text: text,

                senderId: myUsername,

                effect: "",

                customColor: "",

                isSilent: false,

                timestamp: timestamp

            };

            

            // Отправляем на сервер Node.js

            socket.emit('send_message', data);

            

            // Рисуем у себя (оптимистичный UI)

            const myMsg = { ...data, isMine: true };

            allMessages.push(myMsg);

            renderMessages();

            

            input.value = '';

            input.focus();

        }


        function handleEnter(e) {

            if (e.key === 'Enter') sendMessage();

        }
